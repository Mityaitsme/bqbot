# TESTING
1. The database file must be located at: data/quest.db
2. Open .env file and insert this link: DATABASE_URL=sqlite:///data/quest.db
3. Set storage adress in .env in STORAGE_ROOT.
4. Required table names: member, team, riddle, riddle_message, riddle_file (if changed, change the names in config too)
5. Member and team shall not be filled with data, riddle shall be (riddle_message and riddle_file too).
6. To alter the database, enter in cmd: 'sqlite3 data/quest.db'

7. Creating table team:
DROP TABLE IF EXISTS team;
CREATE TABLE team (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL UNIQUE,
  password_hash TEXT NOT NULL,
  start_stage INTEGER NOT NULL DEFAULT 0,
  cur_stage INTEGER NOT NULL DEFAULT 0,
  score INTEGER NOT NULL DEFAULT 0,
  cur_member_id INTEGER,
  stage_call_time INTEGER
);

8. Creating table member:
DROP TABLE IF EXISTS member;
CREATE TABLE member (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  tg_nickname TEXT,
  name TEXT NOT NULL,
  team_id INTEGER NOT NULL,

  FOREIGN KEY (team_id) REFERENCES team(id)
);

9. Creating table riddle:
DROP TABLE IF EXISTS riddle;
CREATE TABLE riddle (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  answer TEXT NOT NULL,
  type TEXT NOT NULL
);

10. Creating table riddle_message (messages for riddles)
DROP TABLE IF EXISTS riddle_message;
CREATE TABLE riddle_message (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  riddle_id INTEGER NOT NULL REFERENCES riddle(id) ON DELETE CASCADE,
  text TEXT NOT NULL
);

11. Creating table riddle_file (files for messages for riddles)
DROP TABLE IF EXISTS riddle_file;
CREATE TABLE riddle_file (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  message_id INTEGER NOT NULL REFERENCES riddle_message(id) ON DELETE CASCADE,
  filename TEXT NOT NULL
);
